第一支 C++ 程式：取得系統資訊

C++ system_info.cpp — 取得電腦基本資訊 
#include <windows.h>
#include <iostream>
#include <string>
using namespace std;

int main() {
    // 取得電腦名稱
    char computerName[256];
    DWORD size = sizeof(computerName);
    GetComputerNameA(computerName, &size);
    cout << "電腦名稱: " << computerName << endl;

    // 取得 Windows 版本資訊
    OSVERSIONINFOEX osInfo = {};
    osInfo.dwOSVersionInfoSize = sizeof(osInfo);
    // 使用 RtlGetVersion（更準確）
    NTSTATUS(WINAPI *RtlGetVersion)(PRTL_OSVERSIONINFOW) =
        (decltype(RtlGetVersion))GetProcAddress(
            GetModuleHandle(L"ntdll"), "RtlGetVersion");

    if (RtlGetVersion) {
        RTL_OSVERSIONINFOW rtlOsInfo = {};
        rtlOsInfo.dwOSVersionInfoSize = sizeof(rtlOsInfo);
        RtlGetVersion(&rtlOsInfo);
        cout << "Windows 版本: "
             << rtlOsInfo.dwMajorVersion << "."
             << rtlOsInfo.dwMinorVersion << endl;
    }

    // 取得記憶體資訊
    MEMORYSTATUSEX memStatus = {};
    memStatus.dwLength = sizeof(memStatus);
    GlobalMemoryStatusEx(&memStatus);
    cout << "記憶體使用率: " << memStatus.dwMemoryLoad << "%" << endl;
    cout << "總實體記憶體: "
         << (memStatus.ullTotalPhys / 1024 / 1024) << " MB" << endl;

    return 0;
}

系統 API 與 Windows 核心概念

掌握 Windows API 的核心概念，這是所有防禦程式的基石。

Windows API 架構

📦

User Mode（使用者模式）

一般應用程式執行的層級。你的防禦程式大部分在這裡運作。

⚙️

Kernel Mode（核心模式）

作業系統核心層級。防毒軟體的驅動程式在這裡，具有最高權限。

🔑

HANDLE（控制代碼）

Windows 用來代表資源（程序、檔案、登錄機碼）的識別碼。

🎯

PID（程序 ID）

每個執行中程序的唯一識別數字，用來定位和操控程序。

錯誤處理：防禦程式的必備技能

系統 API 失敗時，必須使用 GetLastError() 取得錯誤碼。

C++ error_handling.cpp 
#include <windows.h>
#include <iostream>
#include <string>
using namespace std;

// 輔助函數：將錯誤碼轉換成可讀訊息
string GetWinError(DWORD errorCode) {
    LPSTR msgBuf = nullptr;
    FormatMessageA(
        FORMAT_MESSAGE_ALLOCATE_BUFFER |
        FORMAT_MESSAGE_FROM_SYSTEM |
        FORMAT_MESSAGE_IGNORE_INSERTS,
        nullptr,
        errorCode,
        MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT),
        (LPSTR)&msgBuf, 0, nullptr
    );
    string result = msgBuf ? msgBuf : "未知錯誤";
    LocalFree(msgBuf);
    return result;
}

// 嘗試以系統管理員權限開啟程序
HANDLE SafeOpenProcess(DWORD pid) {
    HANDLE hProcess = OpenProcess(
        PROCESS_ALL_ACCESS, FALSE, pid
    );

    if (hProcess == nullptr) {
        DWORD err = GetLastError();
        cerr << "[錯誤] 無法開啟程序 PID=" << pid
             << " 錯誤碼:" << err
             << " (" << GetWinError(err) << ")" << endl;
        return nullptr;
    }
    return hProcess;
}

int main() {
    // 測試取得自身程序的 HANDLE
    DWORD myPid = GetCurrentProcessId();
    cout << "自身 PID: " << myPid << endl;

    HANDLE hSelf = SafeOpenProcess(myPid);
    if (hSelf) {
        cout << "成功取得自身程序控制代碼" << endl;
        CloseHandle(hSelf); // 記得釋放！
    }
    return 0;
}

程序與執行緒監控

學習如何列舉、監控所有執行中的程序——找出病毒的第一步。

列舉所有執行中的程序

C++ process_monitor.cpp — 掃描所有程序 
#include <windows.h>
#include <tlhelp32.h>
#include <iostream>
#include <vector>
#include <string>
using namespace std;

struct ProcessInfo {
    DWORD  pid;
    DWORD  parentPid;
    string name;
    DWORD  threadCount;
};

// 列出所有執行中程序
vector<ProcessInfo> ListAllProcesses() {
    vector<ProcessInfo> result;

    // 建立程序快照
    HANDLE hSnap = CreateToolhelp32Snapshot(
        TH32CS_SNAPPROCESS, 0
    );
    if (hSnap == INVALID_HANDLE_VALUE) return result;

    PROCESSENTRY32 pe32 = {};
    pe32.dwSize = sizeof(pe32);

    // 取得第一個程序
    if (!Process32First(hSnap, &pe32)) {
        CloseHandle(hSnap);
        return result;
    }

    do {
        ProcessInfo info;
        info.pid         = pe32.th32ProcessID;
        info.parentPid   = pe32.th32ParentProcessID;
        info.name        = pe32.szExeFile;
        info.threadCount = pe32.cntThreads;
        result.push_back(info);
    } while (Process32Next(hSnap, &pe32));

    CloseHandle(hSnap);
    return result;
}

// 可疑程序名稱清單（可自行擴充）
bool IsSuspicious(const string& name) {
    vector<string> blacklist = {
        "miner.exe", "keylogger.exe", "cryptolocker.exe",
        "ransomware.exe", "trojan.exe"
        // 實際使用時需要更完整的特徵資料庫
    };
    for (auto& b : blacklist) {
        if (name == b) return true;
    }
    return false;
}

int main() {
    cout << "=== 程序掃描器 ===\n";
    auto procs = ListAllProcesses();
    cout << "共偵測到 " << procs.size() << " 個程序\n\n";

    for (auto& p : procs) {
        bool sus = IsSuspicious(p.name);
        if (sus)
            cout << "[⚠️ 可疑] ";
        cout << "PID:" << p.pid
             << " PPID:" << p.parentPid
             << " Threads:" << p.threadCount
             << " Name:" << p.name << "\n";
    }
    return 0;
}

持續監控：偵測新啟動的程序

C++ process_watcher.cpp — 即時監控新程序 
#include <windows.h>
#include <tlhelp32.h>
#include <iostream>
#include <set>
using namespace std;

set<DWORD> GetCurrentPIDs() {
    set<DWORD> pids;
    HANDLE hSnap = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);
    if (hSnap == INVALID_HANDLE_VALUE) return pids;

    PROCESSENTRY32 pe = { sizeof(pe) };
    if (Process32First(hSnap, &pe))
        do { pids.insert(pe.th32ProcessID); }
        while (Process32Next(hSnap, &pe));
    CloseHandle(hSnap);
    return pids;
}

int main() {
    cout << "[監控啟動] 每秒偵測新程序...\n";
    auto prev = GetCurrentPIDs();

    while (true) {
        Sleep(1000); // 每秒掃描一次
        auto curr = GetCurrentPIDs();

        // 找出新出現的 PID
        for (auto pid : curr) {
            if (prev.find(pid) == prev.end()) {
                cout << "[新程序] PID: " << pid << "\n";
                // TODO: 在這裡加入你的判斷邏輯
            }
        }
        prev = curr;
    }
}

Chapter 04 · 防禦核心

開機啟動與服務常駐

讓你的防護程式比病毒更早啟動，並在後台持續守護系統。

Windows 服務 vs 一般程式的差異

🏃

一般啟動（不推薦）

放在 HKCU\Software\Microsoft\Windows\CurrentVersion\Run，容易被病毒攔截或優先度相同。

⚙️

Windows 服務（推薦）

開機時由 SCM（服務控制管理員）啟動，在用戶登入前就已執行，優先度高。

建立 Windows 服務（核心程式碼）

C++ windows_service.cpp — 防護服務骨架 
#include <windows.h>
#include <winsvc.h>
#include <iostream>
using namespace std;

const wchar_t* SERVICE_NAME = L"GuardianShield";

SERVICE_STATUS        g_ServiceStatus = {};
SERVICE_STATUS_HANDLE g_StatusHandle  = nullptr;
HANDLE                g_StopEvent     = nullptr;

// ── 服務控制處理器 ──
VOID WINAPI ServiceCtrlHandler(DWORD CtrlCode) {
    switch (CtrlCode) {
    case SERVICE_CONTROL_STOP:
        g_ServiceStatus.dwCurrentState = SERVICE_STOP_PENDING;
        SetServiceStatus(g_StatusHandle, &g_ServiceStatus);
        SetEvent(g_StopEvent); // 通知主迴圈停止
        break;
    }
}

// ── 你的防護邏輯放在這裡 ──
void DoProtection() {
    // 每 5 秒執行一次掃描（後續章節會補充）
    Sleep(5000);
}

// ── 服務主函數 ──
VOID WINAPI ServiceMain(DWORD argc, LPWSTR* argv) {
    // 向 SCM 註冊控制處理器
    g_StatusHandle = RegisterServiceCtrlHandler(
        SERVICE_NAME, ServiceCtrlHandler
    );

    g_ServiceStatus.dwServiceType  = SERVICE_WIN32_OWN_PROCESS;
    g_ServiceStatus.dwCurrentState = SERVICE_START_PENDING;
    SetServiceStatus(g_StatusHandle, &g_ServiceStatus);

    // 建立停止事件
    g_StopEvent = CreateEvent(nullptr, TRUE, FALSE, nullptr);

    // 回報「執行中」
    g_ServiceStatus.dwCurrentState  = SERVICE_RUNNING;
    g_ServiceStatus.dwControlsAccepted = SERVICE_ACCEPT_STOP;
    SetServiceStatus(g_StatusHandle, &g_ServiceStatus);

    // ── 防護主迴圈 ──
    while (WaitForSingleObject(g_StopEvent, 0) != WAIT_OBJECT_0) {
        DoProtection(); // 持續執行防護工作
    }

    CloseHandle(g_StopEvent);
    g_ServiceStatus.dwCurrentState = SERVICE_STOPPED;
    SetServiceStatus(g_StatusHandle, &g_ServiceStatus);
}

int main() {
    SERVICE_TABLE_ENTRY ServiceTable[] = {
        { (LPWSTR)SERVICE_NAME, ServiceMain },
        { nullptr, nullptr }
    };
    StartServiceCtrlDispatcher(ServiceTable);
    return 0;
}

檔案系統監視

即時監控檔案變化，偵測勒索軟體的加密行為或病毒植入。

使用 ReadDirectoryChangesW 監視資料夾

C++ file_monitor.cpp — 即時檔案監視 
#include <windows.h>
#include <iostream>
#include <string>
using namespace std;

void MonitorDirectory(const wstring& path) {
    HANDLE hDir = CreateFile(
        path.c_str(),
        FILE_LIST_DIRECTORY,
        FILE_SHARE_READ | FILE_SHARE_WRITE | FILE_SHARE_DELETE,
        nullptr,
        OPEN_EXISTING,
        FILE_FLAG_BACKUP_SEMANTICS,
        nullptr
    );

    if (hDir == INVALID_HANDLE_VALUE) {
        cerr << "無法開啟目錄監視\n";
        return;
    }

    BYTE buffer[65536];
    DWORD bytesReturned;

    wcout << L"[監視中] " << path << L"\n";

    while (true) {
        ReadDirectoryChangesW(
            hDir, buffer, sizeof(buffer),
            TRUE, // 監視子目錄
            FILE_NOTIFY_CHANGE_FILE_NAME |
            FILE_NOTIFY_CHANGE_DIR_NAME  |
            FILE_NOTIFY_CHANGE_LAST_WRITE |
            FILE_NOTIFY_CHANGE_CREATION,
            &bytesReturned, nullptr, nullptr
        );

        FILE_NOTIFY_INFORMATION* fni =
            (FILE_NOTIFY_INFORMATION*)buffer;

        do {
            wstring fileName(
                fni->FileName,
                fni->FileNameLength / sizeof(wchar_t)
            );

            wstring action;
            switch (fni->Action) {
            case FILE_ACTION_ADDED:    action = L"新增"; break;
            case FILE_ACTION_REMOVED:  action = L"刪除"; break;
            case FILE_ACTION_MODIFIED: action = L"修改"; break;
            case FILE_ACTION_RENAMED_NEW_NAME:
                                        action = L"重新命名"; break;
            }

            wcout << L"[" << action << L"] " << fileName << L"\n";

            // ⚠️ 偵測可疑副檔名（勒索軟體常見）
            if (fileName.find(L".locked") != wstring::npos ||
                fileName.find(L".encrypted") != wstring::npos ||
                fileName.find(L".ransom") != wstring::npos) {
                wcerr << L"[🚨警告] 疑似勒索軟體加密行為！\n";
                // TODO: 觸發警報 / 終止程序
            }

            if (fni->NextEntryOffset == 0) break;
            fni = (FILE_NOTIFY_INFORMATION*)
                ((BYTE*)fni + fni->NextEntryOffset);
        } while (true);
    }
    CloseHandle(hDir);
}

int main() {
    // 監視多個重要目錄
    MonitorDirectory(L"C:\\Windows\\System32");
    return 0;
}

Chapter 06 · 防禦核心

登錄檔防護

監控與保護登錄檔，防止病毒修改開機項目或封鎖 CMD/系統管理員。

關鍵登錄機碼：病毒最愛的藏身之處

路徑作用病毒用途
HKLM\SOFTWARE\Microsoft\Windows\CurrentVersion\Run開機自動執行加入自身到開機項
HKLM\SOFTWARE\Microsoft\Windows NT\CurrentVersion\Image File Execution Options程式執行重導向攔截 cmd.exe、regedit.exe
HKLM\SAM\SAM\Domains\Account使用者帳號資料鎖定/刪除管理員帳號
HKLM\SYSTEM\CurrentControlSet\Services服務清單新增惡意服務

掃描開機項目

C++ registry_scan.cpp 
#include <windows.h>
#include <iostream>
#include <string>
using namespace std;

void ScanRunKeys() {
    const wchar_t* runKeys[] = {
        L"SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\Run",
        L"SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\RunOnce"
    };

    for (auto* keyPath : runKeys) {
        HKEY hKey;
        if (RegOpenKeyEx(HKEY_LOCAL_MACHINE, keyPath,
                0, KEY_READ, &hKey) != ERROR_SUCCESS)
            continue;

        wcout << L"\n[掃描] HKLM\\" << keyPath << L"\n";

        DWORD index = 0;
        wchar_t valueName[256], valueData[1024];
        DWORD nameLen, dataLen, type;

        while (true) {
            nameLen = 256; dataLen = sizeof(valueData);
            LONG ret = RegEnumValue(
                hKey, index++, valueName, &nameLen,
                nullptr, &type,
                (LPBYTE)valueData, &dataLen
            );
            if (ret != ERROR_SUCCESS) break;

            wcout << L"  名稱: " << valueName << L"\n";
            wcout << L"  路徑: " << valueData << L"\n";

            // 偵測可疑路徑（在 Temp 目錄的可執行檔往往可疑）
            wstring data(valueData);
            if (data.find(L"\\Temp\\") != wstring::npos ||
                data.find(L"\\AppData\\Local\\Temp") != wstring::npos) {
                wcout << L"  [⚠️ 可疑] 路徑在 Temp 目錄！\n";
            }
        }
        RegCloseKey(hKey);
    }
}

// 偵測 IFEO（CMD/Task Manager 被攔截的病毒手法）
void CheckIFEO() {
    const wchar_t* targets[] = {
        L"cmd.exe", L"taskmgr.exe",
        L"regedit.exe", L"msconfig.exe"
    };
    const wstring basePath =
        L"SOFTWARE\\Microsoft\\Windows NT\\CurrentVersion\\Image File Execution Options\\";

    wcout << L"\n[IFEO 掃描] 確認系統工具是否被攔截:\n";
    for (auto* t : targets) {
        HKEY hKey;
        wstring fullPath = basePath + t;
        if (RegOpenKeyEx(HKEY_LOCAL_MACHINE,
                fullPath.c_str(), 0, KEY_READ, &hKey)
                == ERROR_SUCCESS) {
            wchar_t debugger[512];
            DWORD len = sizeof(debugger);
            if (RegQueryValueEx(hKey, L"Debugger",
                    nullptr, nullptr,
                    (LPBYTE)debugger, &len)
                    == ERROR_SUCCESS) {
                wcout << L"  [🚨攔截] " << t
                      << L" 被重導向至: " << debugger << L"\n";
                // TODO: 自動刪除這個惡意 Debugger 值
            } else {
                wcout << L"  [✅正常] " << t << L"\n";
            }
            RegCloseKey(hKey);
        } else {
            wcout << L"  [✅正常] " << t << L"\n";
        }
    }
}

int main() {
    ScanRunKeys();
    CheckIFEO();
    return 0;
}
🔴 你的 CMD 被鎖住的原因就是這個！
IFEO（Image File Execution Options）是病毒最常用來封鎖 cmd.exe、taskmgr.exe 的方法。這段程式可以直接偵測並揭露這種攻擊。

Chapter 07 · 進階篇

程序注入偵測

偵測 DLL 注入、程式碼注入等進階攻擊——這是現代惡意軟體最常用的技術。

程序注入的常見類型

💊

DLL Injection

將惡意 DLL 載入到合法程序（如 explorer.exe）中執行，借殼逃避偵測。

🧠

Process Hollowing

啟動合法程序後，清空其記憶體並替換成惡意程式碼。

🪝

SetWindowsHookEx

在其他程序中安裝鍵盤/滑鼠 Hook，記錄使用者輸入。

📝

WriteProcessMemory

直接將 Shellcode 寫入另一個程序的記憶體並執行。

掃描程序已載入的 DLL 模組

C++ dll_scanner.cpp — 偵測可疑 DLL 
#include <windows.h>
#include <tlhelp32.h>
#include <psapi.h>
#include <iostream>
#include <string>
#include <vector>
#pragma comment(lib, "psapi.lib")
using namespace std;

// 取得程序所有已載入的 DLL 清單
vector<wstring> GetProcessModules(DWORD pid) {
    vector<wstring> modules;

    HANDLE hProc = OpenProcess(
        PROCESS_QUERY_INFORMATION | PROCESS_VM_READ,
        FALSE, pid
    );
    if (!hProc) return modules;

    HMODULE hMods[1024];
    DWORD cbNeeded;

    if (EnumProcessModules(hProc, hMods, sizeof(hMods), &cbNeeded)) {
        DWORD count = cbNeeded / sizeof(HMODULE);
        for (DWORD i = 0; i < count; i++) {
            wchar_t modName[MAX_PATH];
            if (GetModuleFileNameEx(
                    hProc, hMods[i], modName, MAX_PATH))
                modules.push_back(modName);
        }
    }
    CloseHandle(hProc);
    return modules;
}

// 判斷 DLL 路徑是否可疑
bool IsSuspiciousDll(const wstring& path) {
    // 不在 System32 或 Program Files 的 DLL 值得注意
    if (path.find(L"\\Temp\\") != wstring::npos) return true;
    if (path.find(L"\\AppData\\Local\\Temp") != wstring::npos) return true;
    if (path.find(L"\\Downloads\\") != wstring::npos) return true;
    return false;
}

void ScanProcessForInjection(DWORD pid, const string& procName) {
    auto mods = GetProcessModules(pid);
    for (auto& mod : mods) {
        if (IsSuspiciousDll(mod)) {
            cout << "[🚨注入偵測] 程序: " << procName
                 << " (PID:" << pid << ")\n";
            wcout << L"  可疑 DLL: " << mod << L"\n";
        }
    }
}

int main() {
    cout << "[DLL 注入掃描器]\n";

    HANDLE hSnap = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);
    PROCESSENTRY32 pe = { sizeof(pe) };
    if (Process32First(hSnap, &pe)) {
        do {
            ScanProcessForInjection(pe.th32ProcessID, pe.szExeFile);
        } while (Process32Next(hSnap, &pe));
    }
    CloseHandle(hSnap);
    return 0;
}

Chapter 08 · 進階篇

強制終止惡意程序

偵測到威脅後，如何在病毒的 Watchdog 反應前，搶先終止它。

提升權限：啟用 SeDebugPrivilege

C++ privilege.cpp — 取得除錯權限 
#include <windows.h>
#include <iostream>
using namespace std;

// 啟用指定的權限（對抗受保護的惡意程序）
bool EnablePrivilege(const wchar_t* privName) {
    HANDLE hToken;
    if (!OpenProcessToken(
            GetCurrentProcess(),
            TOKEN_ADJUST_PRIVILEGES | TOKEN_QUERY,
            &hToken))
        return false;

    TOKEN_PRIVILEGES tp = {};
    LookupPrivilegeValue(nullptr, privName, &tp.Privileges[0].Luid);
    tp.PrivilegeCount = 1;
    tp.Privileges[0].Attributes = SE_PRIVILEGE_ENABLED;

    bool ok = AdjustTokenPrivileges(
        hToken, FALSE, &tp, sizeof(tp), nullptr, nullptr
    ) && GetLastError() == ERROR_SUCCESS;

    CloseHandle(hToken);
    return ok;
}

// 以提升的權限終止程序
bool KillProcess(DWORD pid) {
    // 先嘗試啟用除錯權限
    EnablePrivilege(SE_DEBUG_NAME);

    HANDLE hProc = OpenProcess(
        PROCESS_TERMINATE, FALSE, pid
    );
    if (!hProc) {
        cerr << "[錯誤] 無法開啟程序 PID=" << pid << "\n";
        return false;
    }

    bool ok = TerminateProcess(hProc, 1) != 0;
    CloseHandle(hProc);
    return ok;
}

// 同時終止多個相互守護的程序
void KillWatchdogPair(DWORD pid1, DWORD pid2) {
    cout << "[⚔️] 同步終止 Watchdog 程序對...\n";
    // 先暫停兩個程序（讓它們無法重啟對方）
    HANDLE h1 = OpenProcess(PROCESS_ALL_ACCESS, FALSE, pid1);
    HANDLE h2 = OpenProcess(PROCESS_ALL_ACCESS, FALSE, pid2);

    // 掛起（暫停）兩個程序
    using fnNtSuspend = LONG(NTAPI*)(HANDLE);
    auto NtSuspendProcess = (fnNtSuspend)GetProcAddress(
        GetModuleHandle(L"ntdll"), "NtSuspendProcess"
    );
    if (NtSuspendProcess) {
        if (h1) NtSuspendProcess(h1); // 先凍結
        if (h2) NtSuspendProcess(h2); // 再凍結
    }
    // 再終止
    if (h1) { TerminateProcess(h1, 1); CloseHandle(h1); }
    if (h2) { TerminateProcess(h2, 1); CloseHandle(h2); }
    cout << "[✅] Watchdog 程序對已終止\n";
}

int main() {
    // 範例：終止 PID 1234 和 5678（病毒的 Watchdog 對）
    KillWatchdogPair(1234, 5678);
    return 0;
}
✅ 這就是對抗「刪除後自動重啟」的核心技術！
先用 NtSuspendProcess 同時凍結兩個互相守護的程序，讓它們無法重啟對方，再一起終止。這破解了 Watchdog 互守機制。

網路流量監控

偵測病毒的 C2（Command and Control）連線，阻止資料外洩。

使用 IP Helper API 列舉連線

C++ network_monitor.cpp 
#include <windows.h>
#include <iphlpapi.h>
#include <winsock2.h>
#include <ws2tcpip.h>
#include <iostream>
#include <set>
#pragma comment(lib, "iphlpapi.lib")
#pragma comment(lib, "ws2_32.lib")
using namespace std;

// 已知 C2 / 惡意 IP 黑名單（示範用）
set<string> maliciousIPs = {
    "192.168.1.100", // 替換成實際黑名單
    "10.0.0.99"
};

string IpToString(DWORD ip) {
    struct in_addr addr;
    addr.s_addr = ip;
    return inet_ntoa(addr);
}

void MonitorConnections() {
    DWORD size = 0;
    GetExtendedTcpTable(nullptr, &size, FALSE,
        AF_INET, TCP_TABLE_OWNER_PID_ALL, 0);

    vector<BYTE> buf(size);
    auto* table = (MIB_TCPTABLE_OWNER_PID*)buf.data();

    if (GetExtendedTcpTable(table, &size, FALSE,
            AF_INET, TCP_TABLE_OWNER_PID_ALL, 0)
            != NO_ERROR) return;

    cout << "=== 網路連線掃描 ===\n";
    for (DWORD i = 0; i < table->dwNumEntries; i++) {
        auto& row = table->table[i];
        if (row.dwState != MIB_TCP_STATE_ESTAB) continue;

        string remoteIP = IpToString(row.dwRemoteAddr);
        int    remotePort = ntohs((u_short)row.dwRemotePort);
        DWORD  pid = row.dwOwningPid;

        cout << "PID:" << pid
             << " → " << remoteIP << ":" << remotePort;

        if (maliciousIPs.count(remoteIP)) {
            cout << "  [🚨 C2 連線！]";
            // TODO: 封鎖 + 終止對應程序
        }
        cout << "\n";
    }
}

int main() {
    WSADATA wsa;
    WSAStartup(MAKEWORD(2,2), &wsa);
    MonitorConnections();
    WSACleanup();
    return 0;
}

C++ guardian_shield.cpp — 完整防護框架 
#include <windows.h>
#include <thread>
#include <mutex>
#include <queue>
#include <atomic>
#include <iostream>
#include <fstream>
#include <chrono>
using namespace std;

// ── 全域狀態 ──
atomic<bool> g_running{ true };
mutex         g_logMutex;
mutex         g_queueMutex;
queue<string> g_eventQueue;

// ── 日誌系統 ──
void Log(const string& msg, bool alert = false) {
    auto now = chrono::system_clock::now();
    auto t   = chrono::system_clock::to_time_t(now);

    lock_guard<mutex> lk(g_logMutex);
    if (alert) cout << "[🚨ALERT] ";
    else       cout << "[INFO]    ";
    cout << msg << "\n";

    ofstream logFile("guardian.log", ios::app);
    logFile << ctime(&t) << (alert?"ALERT: ":"INFO:  ") << msg << "\n";
}

// ── 事件推送 ──
void PushEvent(const string& ev) {
    lock_guard<mutex> lk(g_queueMutex);
    g_eventQueue.push(ev);
}

// ── 程序監控執行緒 ──
void ProcessMonitorThread() {
    Log("程序監控執行緒啟動");
    // 整合 Chapter 3 的程式碼
    while (g_running) {
        // ListAllProcesses() + 可疑檢查 + PushEvent
        Sleep(2000);
    }
}

// ── 檔案監控執行緒 ──
void FileMonitorThread() {
    Log("檔案系統監控執行緒啟動");
    // 整合 Chapter 5 的 MonitorDirectory
    while (g_running) {
        // ReadDirectoryChangesW 已是阻塞式呼叫，在內部等待
        Sleep(100);
    }
}

// ── 登錄檔監控執行緒 ──
void RegistryMonitorThread() {
    Log("登錄檔監控執行緒啟動");
    HANDLE hKey;
    RegOpenKeyEx(HKEY_LOCAL_MACHINE,
        L"SOFTWARE\\Microsoft\\Windows\\CurrentVersion\\Run",
        0, KEY_NOTIFY, &hKey);

    while (g_running) {
        // RegNotifyChangeKeyValue 等待登錄檔變更
        LONG ret = RegNotifyChangeKeyValue(
            hKey, TRUE,
            REG_NOTIFY_CHANGE_LAST_SET |
            REG_NOTIFY_CHANGE_NAME,
            nullptr, FALSE
        );
        if (ret == ERROR_SUCCESS) {
            PushEvent("ALERT:登錄檔 Run 機碼被修改！");
        }
    }
    RegCloseKey(hKey);
}

// ── 主事件處理迴圈 ──
void EventDispatcher() {
    while (g_running) {
        string ev;
        {
            lock_guard<mutex> lk(g_queueMutex);
            if (!g_eventQueue.empty()) {
                ev = g_eventQueue.front();
                g_eventQueue.pop();
            }
        }
        if (!ev.empty()) {
            bool isAlert = ev.find("ALERT") != string::npos;
            Log(ev, isAlert);
            if (isAlert) {
                // 觸發聲音警報
                MessageBeep(MB_ICONEXCLAMATION);
            }
        }
        Sleep(100);
    }
}

int main() {
    cout << R"(
  ╔═══════════════════════════════════╗
  ║   Guardian Shield v1.0            ║
  ║   C++ 電腦防禦系統               ║
  ╚═══════════════════════════════════╝
)";

    Log("Guardian Shield 防護系統啟動");

    // 啟動各監控執行緒
    thread t1(ProcessMonitorThread);
    thread t2(FileMonitorThread);
    thread t3(RegistryMonitorThread);
    thread t4(EventDispatcher);

    // 等待 Ctrl+C 或服務停止信號
    cout << "防護系統執行中...按 Enter 停止\n";
    cin.get();
    g_running = false;

    t1.join(); t2.join();
    t3.join(); t4.join();
    Log("Guardian Shield 已停止");
    return 0;
}
